const express = require('express')
const app = express()
const mongoose = require('mongoose')
const bodyParser = require('body-parser')

app.use(bodyParser.urlencoded({extended: true}));

mongoose.connect('mongodb+srv://andrei:andrei@cluster0.rkr6oej.mongodb.net/nodePlayground');

//create schema
const studentsSchema = {
    nume: String, 
    puncte: String
}

const Student = mongoose.model('Student', studentsSchema)

app.get("/", function(req, res) {
    res.sendFile(__dirname + '/SSS.html')
})

app.post('/', function(req, res) {
    const newStudent = new Student({
      nume: req.body.nume,
      puncte: req.body.puncte,
    });
  
    
    console.log(`nume: ${req.body.nume}, puncte: ${req.body.puncte}`)

    newStudent.save()
    res.redirect('/');
  });

app.listen(3000, function() {
    console.log('serveru alearga')
})
